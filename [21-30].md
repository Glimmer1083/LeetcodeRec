### 21、合并有序链表

递归解法：递归下一步的输入变量想错了，十分的蠢了！

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, l1: ListNode, l2: ListNode) -> ListNode:
        if l1 == None:
            return l2
        else l2 == None:
            return l1
        if l1.val <= l2.val:
            l1.next = self.mergeTwoLists(l1.next, l2)
            return l1
        elif:
            l2.next = self.mergeTwoLists(l2.next, l1)
            return l2
```

### 27、移除元素

笨办法

```python
class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        i = 0
        while i < len(nums):
            if nums[i] == val:
                del nums[i]
                i -= 1
            i += 1
        return len(nums)
```

```python
class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        sign = 1
        if divisor < 0:
            sign *= -1
            divisor = -divisor
        if dividend < 0:
            sign *= -1
            dividend = - dividend
        i = 0
        res = 0
        if divisor > dividend:
            return 0
        elif divisor <= dividend and divisor << 1 > dividend:
            res = 1*sign
        elif divisor << 1 == dividend:
            res = 2*sign
        else:
            while True:
                if (divisor << i) <= dividend  and (divisor << i + 1) > dividend:
                    break
                elif (divisor << i + 1) == dividend:
                    i += 1
                    break
                else:
                    i += 1
            res = ((1 << i) + self.divide(dividend - (divisor << i), divisor))*sign
        if res > (1 << 31) - 1:
            return (1 <<31) - 1
        elif res <= -2**31:
            return -2**31
        else:
            return res
```
